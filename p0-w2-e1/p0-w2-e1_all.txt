---------kabisat algoritma----------

1.menentukan tahun
2.membagi tahun dengan 4
  2.a apabila tahun dibagi 4 mempunyai hasil genap dan tahun di bagi 100 mempunyai hasil ganjil
      maka teriak bukan kabisat
  2.b apabila tahun dibagi 4 mempunyai hasil genap dan tahun di bagi 100 mempunyai hasil genap
      dan tahun di bagi 400 mempunyai hasil genap
      maka teriak kabisat

---------kabisat pseudocode-------

SET  y = any
IF y MOD 4 EQUAL 0 && y MOD 100 NOT EQUAL 0 
DO DISPLAY = 'TAHUN KABISAT'
ELSE IF y MOD 4 EQUAL 0 && y MOD 100 EQUAL 0 && y MOD 400 EQUAL 0
DO DISPLAY = 'BUKAN TAHUN KABISAT'
END IF

---------loundry algoritma----------

1.membuka penutup mesin cuci
2.memasukan baju satu-persatu dan menghitung jumlah total baju
  2.a apabila jumlah baju yang dihitung kurang dari 20, kembali 
      ketahap 2
3.menutup penutup mesin cuci
4.menghidupkan mesin cuci

---------loundry pseudocode---------

SET b = 0
WHILE b LESSTHAN 20
IF b LESSTHAN 20
DO DISPLAY 'total baju'PLUS b
ELSE
DO DISPLAY 'baju siap di cuci'
END IF	
b PLUS 1
END WHILE
